# Voice extras: install ON TOP of requirements.txt for voice notes, the
# hold-to-talk mic, and Call mode:
#
#   pip install -r requirements.txt -r requirements-voice.txt
#
# All of these are lazy-imported: without them the app still boots and runs
# text chat + memory; only the /stt, /tts and /ws/call endpoints report that
# voice is unavailable.

# Speech-to-text (Whisper small, int8, GPU/CPU) and text-to-speech (Kokoro-82M,
# CPU). Kokoro pulls torch; faster-whisper pulls ctranslate2.
faster-whisper==1.2.1
kokoro==0.9.4
soundfile==0.14.0       # WAV encoding for TTS output
av==18.0.0              # audio decoding (webm/opus) for STT
# Note: kokoro needs espeak-ng installed on the system for phonemization.
#   Windows: install espeak-ng (github.com/espeak-ng/espeak-ng/releases)
#   macOS:   brew install espeak-ng
#   Linux:   apt-get install espeak-ng
# GPU STT also needs CUDA + cuDNN (via a CUDA-enabled PyTorch); CPU works without.
